import { X, ListChecks, Save, Copy, Check } from 'lucide-react';
import { Button } from '@/components/ui/button';
import { useState } from 'react';

interface MenuDrawerProps {
  isOpen: boolean;
  onClose: () => void;
  onNavigate: (view: 'habits' | 'save') => void;
  uniqueId: string | null;
  onGenerateId: () => string;
}

const MenuDrawer = ({ isOpen, onClose, onNavigate, uniqueId, onGenerateId }: MenuDrawerProps) => {
  const [showSaveModal, setShowSaveModal] = useState(false);
  const [generatedId, setGeneratedId] = useState<string | null>(uniqueId);
  const [copied, setCopied] = useState(false);

  if (!isOpen) return null;

  const handleSaveProgress = () => {
    if (!generatedId) {
      const newId = onGenerateId();
      setGeneratedId(newId);
    }
    setShowSaveModal(true);
  };

  const handleCopyId = async () => {
    if (generatedId) {
      await navigator.clipboard.writeText(generatedId);
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
    }
  };

  return (
    <div className="fixed inset-0 z-50">
      <div 
        className="absolute inset-0 bg-background/80 backdrop-blur-sm"
        onClick={onClose}
      />
      
      <div className="absolute top-0 right-0 h-full w-80 max-w-full bg-card border-l border-border p-6 animate-slide-up">
        <div className="flex items-center justify-between mb-8">
          <h2 className="text-lg font-semibold text-foreground">Menú</h2>
          <button
            onClick={onClose}
            className="w-10 h-10 rounded-xl bg-muted flex items-center justify-center text-muted-foreground hover:text-foreground transition-colors"
          >
            <X className="w-5 h-5" />
          </button>
        </div>

        <nav className="space-y-2">
          <button
            onClick={() => {
              onNavigate('habits');
              onClose();
            }}
            className="w-full flex items-center gap-3 px-4 py-3 rounded-xl bg-muted/50 text-foreground hover:bg-muted transition-colors"
          >
            <ListChecks className="w-5 h-5 text-primary" />
            Mis hábitos
          </button>

          <button
            onClick={handleSaveProgress}
            className="w-full flex items-center gap-3 px-4 py-3 rounded-xl bg-muted/50 text-foreground hover:bg-muted transition-colors"
          >
            <Save className="w-5 h-5 text-primary" />
            Guardar progreso
          </button>
        </nav>

        {showSaveModal && (
          <div className="mt-8 p-4 bg-muted rounded-xl">
            <p className="text-sm text-muted-foreground mb-3">
              Tu ID único para recuperar tu progreso:
            </p>
            <div className="flex items-center gap-2">
              <code className="flex-1 bg-background px-3 py-2 rounded-lg text-xs text-foreground font-mono break-all">
                {generatedId}
              </code>
              <Button
                size="icon"
                variant="ghost"
                onClick={handleCopyId}
                className="shrink-0"
              >
                {copied ? (
                  <Check className="w-4 h-4 text-primary" />
                ) : (
                  <Copy className="w-4 h-4" />
                )}
              </Button>
            </div>
            <p className="text-xs text-muted-foreground mt-3">
              Guarda este ID para recuperar tu progreso en otro dispositivo.
            </p>
          </div>
        )}
      </div>
    </div>
  );
};

export default MenuDrawer;
